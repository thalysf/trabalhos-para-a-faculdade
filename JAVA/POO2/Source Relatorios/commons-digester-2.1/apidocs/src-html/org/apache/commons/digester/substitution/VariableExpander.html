020    package org.apache.commons.digester.substitution;
021    
022    /**
023     * <p>An Interface describing a class capable of expanding strings which
024     * may contain variable references. The exact syntax of the "reference",
025     * and the mechanism for determining the corresponding value to be used
026     * is up to the concrete implementation.</p>
027     *
028     * @since 1.6
029     */
030    public interface VariableExpander {
031        /**
032         * Return the input string with any variables replaced by their
033         * corresponding value. If there are no variables in the string,
034         * then the input parameter is returned unaltered.
035         */
036        public String expand(String param);
037    }
