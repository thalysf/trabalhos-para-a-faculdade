018    package org.apache.commons.digester.annotations.providers;
019    
020    import java.lang.reflect.Method;
021    
022    import org.apache.commons.digester.SetRootRule;
023    import org.apache.commons.digester.annotations.AnnotationRuleProvider;
024    import org.apache.commons.digester.annotations.rules.SetRoot;
025    
026    /**
027     * Provides instances of {@link SetRootRule}.
028     *
029     * @since 2.1
030     */
031    public final class SetRootRuleProvider implements AnnotationRuleProvider<SetRoot, Method, SetRootRule>{
032    
033        private String methodName;
034    
035        private String paramType;
036    
037        /**
038         * {@inheritDoc}
039         */
040        public void init(SetRoot annotation, Method element) {
041            this.methodName = element.getName();
042            this.paramType = element.getParameterTypes()[0].getName();
043        }
044    
045        /**
046         * {@inheritDoc}
047         */
048        public SetRootRule get() {
049            return new SetRootRule(this.methodName, this.paramType);
050        }
051    
052    }
